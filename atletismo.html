Entrena y Compite con Nosotros

La sección de atletismo del CCRD Ferroviario abarca desde corredores populares de ruta hasta atletas de pista. Nuestro objetivo es fomentar la salud, la superación personal y el entrenamiento en grupo, participando en el circuito de carreras de Sevilla y medias maratones andaluzas.

LÓGICA INTERACTIVA DEL CALENDARIO ========================================================== */ const eventosClub = { '2026-4-28': '🏃‍♂️ CCRD: Reunion de Junta Directiva a las 18:30 en la sede social.', '2026-1-15': '🎯 CCRD: Asamblea ordinaria de socios a las 19:00 en la sede social.', }; let fechaActual = new Date(); const nombresMeses = ["Enero", "Febrero", "Marzo", "Abril", "Mayo", "Junio", "Julio", "Agosto", "Septiembre", "Octubre", "Noviembre", "Diciembre"]; function renderizarCalendario() { const año = fechaActual.getFullYear(); const mes = fechaActual.getMonth(); const textoMesAño = document.getElementById('calendar-month-year'); if (!textoMesAño) return; textoMesAño.innerText = `${nombresMeses[mes]} ${año}`; const primerDiaMes = new Date(año, mes, 1).getDay(); const lunesIndex = primerDiaMes === 0 ? 6 : primerDiaMes - 1; const diasTotalesMes = new Date(año, mes + 1, 0).getDate(); const gridDias = document.getElementById('calendar-days'); if (!gridDias) return; gridDias.innerHTML = ''; for (let i = 0; i < lunesIndex; i++) { const divVacio = document.createElement('div'); divVacio.classList.add('calendar-day', 'empty'); gridDias.appendChild(divVacio); } for (let dia = 1; dia <= diasTotalesMes; dia++) { const elementoDia = document.createElement('div'); elementoDia.classList.add('calendar-day'); elementoDia.innerText = dia; const claveFecha = `${año}-${mes}-${dia}`; if (eventosClub[claveFecha]) { elementoDia.classList.add('has-event'); } const hoy = new Date(); if (dia === hoy.getDate() && mes === hoy.getMonth() && año === hoy.getFullYear()) { elementoDia.classList.add('today'); } elementoDia.addEventListener('click', () => { abrirModalInformacion(dia, nombresMeses[mes], año, eventosClub[claveFecha]); }); gridDias.appendChild(elementoDia); } } function abrirModalInformacion(dia, mesTxt, año, info) { const modal = document.getElementById('event-modal'); document.getElementById('modal-date-title').innerText = `${dia} de ${mesTxt} de ${año}`; if (info) { document.getElementById('modal-info-text').innerHTML = info; } else { document.getElementById('modal-info-text').innerHTML = 'ℹ️ No hay eventos programados o modificaciones para este día. Las secciones deportivas mantienen sus horarios habituales.'; } if (modal) modal.classList.add('active'); } const btnCloseModal = document.querySelector('.modal-close-btn'); if (btnCloseModal) { btnCloseModal.addEventListener('click', () => { document.getElementById('event-modal').classList.remove('active'); }); } window.addEventListener('click', (e) => { const modal = document.getElementById('event-modal'); if (e.target === modal) modal.classList.remove('active'); }); const btnPrevMonth = document.getElementById('prev-month-btn'); const btnNextMonth = document.getElementById('next-month-btn'); if (btnPrevMonth) { btnPrevMonth.addEventListener('click', () => { fechaActual.setMonth(fechaActual.getMonth() - 1); renderizarCalendario(); }); } if (btnNextMonth) { btnNextMonth.addEventListener('click', () => { fechaActual.setMonth(fechaActual.getMonth() + 1); renderizarCalendario(); }); } // Inicialización segura de ambos componentes mostrarSlides(slideIndex); renderizarCalendario(); });
